#!/bin/bash

BAM_DIR="./bam_files"

OUT_DIR="./sorted_bams"

mkdir -p $OUT_DIR

for i in $BAM_DIR/*.bam; do

base=$(basename $i .bam)

samtools sort -@ 4 $i -o $OUT_DIR/$base.sorted.bam &

samtools index $OUT_DIR/$base.sorted.bam &

done

wait

echo "All BAM files processed."
